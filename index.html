Quickstart

Get up and running in seconds using Python's asyncio.

python
import asyncio
from SaavnAPI import SaavnClient

async def main():
    async with SaavnClient() as client:
        results = await client.search_songs("Kesariya")
        for song in results:
            print(song.title, "-", song.download_url)

asyncio.run(main())
💡Use async with SaavnClient() to automatically manage the HTTP session lifecycle.

API Reference

All methods are coroutines — must be awaited inside an async function.

class SaavnClient

The main client. Manages the underlying aiohttp session.

python
from SaavnAPI import SaavnClient

# Recommended: context manager

# Manual session management

await client.start()
await client.close()

Examples

🤖 Telegram Music Bot

python
from SaavnAPI import SaavnClient

async def handle_song_request(query: str) -> str:
    async with SaavnClient() as client:
        songs = await client.search_songs(query, limit=1)
        if not songs:
            return "No songs found."
        s = songs[0]
        return f"🎵 {s.title}\n👤 {s.artist}\n💿 {s.album}\n⏱ {s.duration}s\n🔗 {s.download_url}"

📥 Playlist Downloader

python
import asyncio
from SaavnAPI import SaavnClient

async def download_playlist(playlist_id: str):
    async with SaavnClient() as client:
        pl = await client.get_playlist(playlist_id)
        print(f"📋 {pl.name} ({pl.song_count} songs)")
        for i, song in enumerate(pl.songs, 1):
            print(f"  [{i}] {song.title} - {song.artist}")
            print(f"      {song.download_url}")

asyncio.run(download_playlist("159144718"))

🎤 Lyrics Fetcher

python
import asyncio
from SaavnAPI import SaavnClient

async def get_lyrics(song_name: str):
    async with SaavnClient() as client:
        results = await client.search_songs(song_name, limit=1)
        if results:
            lyrics = await client.get_lyrics(results[0].id)
            print(f"🎵 {results[0].title}")
            print("-" * 40)
            print(lyrics)

asyncio.run(get_lyrics("Tum Hi Ho"))
